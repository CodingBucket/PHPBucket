 *** Laravel Artisan Commands ***
===================================


-----------------------------------
 # Composer version check
-----------------------------------

	CMD: composer

-----------------------------------
 # Laravel Project create
-----------------------------------

    CMD: composer create-project laravel/laravel blog "5.2.*"

----------------------------------- 
 # Laravel artisan comaands show
----------------------------------- 
 	
	CMD: php artisan

-----------------------------------   
 # Create Controller
-----------------------------------

    CMD: php artisan make:controller PhotoController

-----------------------------------    
 # Create restfull resource controller
-----------------------------------

    CMD: php artisan make:controller PhotoController --resource

-----------------------------------
 # Create Model
-----------------------------------

    CMD: php artisan make:model User -m  [-m create migration file]		
    
-----------------------------------    
 # Create Middleware
----------------------------------- 

    CMD: php artisan make:middleware AgeMiddleware
    
-----------------------------------     
 # To run database migration
----------------------------------- 

    CMD: php artisan migrate

-----------------------------------     
 # Rollback migration
----------------------------------- 

    CMD: php artisan migrate:rollback
    
-----------------------------------     
 # Auth create
----------------------------------- 

    CMD: php artisan make:auth
